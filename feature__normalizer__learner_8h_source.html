00083 
00084   for (int feature_index = 0;
00085        feature_index < data_set.feature_count(); ++feature_index) {
00086     double delta = input_max[feature_index] - input_min[feature_index];
00087     double coefficient, shift;
00088     if (utility::DoubleEqual(delta, 0)) {
00089       coefficient = 0.0;
00090       shift = (max_ - min_) / 2;
00091     } else {
00092       coefficient = (max_ - min_) / delta;
00093       shift = min_ - coefficient * input_min[feature_index];
00094     }
00095     per_feature_linear_converter->set_factor(feature_index, coefficient);
00096     per_feature_linear_converter->set_shift(feature_index, shift);
